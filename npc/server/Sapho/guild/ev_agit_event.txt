// WoE Start/Stop times
//======================================
- script Agit_Event -1,{
	end;

// for Tues(2), Thurs(4)
OnClock2100:	//start time
OnClock2200:	//end time
OnClock2210:	//end time
OnClock2220:	//end time
OnClock2230:	//end time
OnClock2240:	//end time
OnClock2245:	//end time
OnClock2250:	//end time
OnClock2255:	//end time
OnClock2300:	//end time
OnClock2305:	//end time

// for Sat(6)
OnClock1700:	//start time
OnClock1800:	//end time
OnClock1810:	//end time
OnClock1820:	//end time
OnClock1830:	//end time
OnClock1840:	//end time
OnClock1845:	//end time
OnClock1850:	//end time
OnClock1855:	//end time
OnClock1900:	//end time
OnClock1905:	//end time

OnAgitInit:
// starting time checks
	if (agitcheck(0) != 0) goto Check_end;
	if (gettime(4) == 2 && (gettime(3) >= 21 && gettime(3) < 22)) goto L_Start;
	if (gettime(4) == 4 && (gettime(3) >= 21 && gettime(3) < 22)) goto L_Start;
	if (gettime(4) == 6 && (gettime(3) >= 17 && gettime(3) < 18)) goto L_Start;
	end;

// end time checks
Check_end:
	if (gettime(4) == 2 && (gettime(3) == 22 || gettime(3) == 23)) goto L_End;
	if (gettime(4) == 4 && (gettime(3) == 22 || gettime(3) == 23)) goto L_End;
	if (gettime(4) == 6 && (gettime(3) == 18 || gettime(3) == 19)) goto L_End;
	end;

L_End:
	// always end on final clock
	// 1 chance of 'number of clocks' to stop WoE
	if (((gettime(3) == 22 || gettime(3) == 18) && gettime(2) ==  0) && rand(10) != 1) end;
	if (((gettime(3) == 22 || gettime(3) == 18) && gettime(2) == 10) && rand( 9) != 1) end;
	if (((gettime(3) == 22 || gettime(3) == 18) && gettime(2) == 20) && rand( 8) != 1) end;
	if (((gettime(3) == 22 || gettime(3) == 18) && gettime(2) == 30) && rand( 7) != 1) end;
	if (((gettime(3) == 22 || gettime(3) == 18) && gettime(2) == 40) && rand( 6) != 1) end;
	if (((gettime(3) == 22 || gettime(3) == 18) && gettime(2) == 45) && rand( 5) != 1) end;
	if (((gettime(3) == 22 || gettime(3) == 18) && gettime(2) == 50) && rand( 4) != 1) end;
	if (((gettime(3) == 22 || gettime(3) == 18) && gettime(2) == 55) && rand( 3) != 1) end;
	if (((gettime(3) == 23 || gettime(3) == 19) && gettime(2) ==  0) && rand( 2) != 1) end;

	Announce "La Guerre de l'Emperium est terminée! Signez le traité de paix!",8;
	AgitEnd;
	end;

L_Start:
	Announce "La Guerre de l'Emperium a commencée! Battez-vous pour vos terres!",8;
	AgitStart;
	end;
}
