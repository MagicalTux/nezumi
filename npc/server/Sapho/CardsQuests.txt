        if (getiteminfo(@item7,10)>0) set @slot$,"["+getiteminfo(@item7,10)+"]";
        if (@quantite7>0) mes "- ^5555FF"+@quantite7+"^000000 "+getitemname(@item7)+@slot$;
        set @slot$,"";
        if (getiteminfo(@item8,10)>0) set @slot$,"["+getiteminfo(@item8,10)+"]";
        if (@quantite8>0) mes "- ^5555FF"+@quantite8+"^000000 "+getitemname(@item8)+@slot$;
        next;
        mes "["+getarg(0)+"]";
        mes "Si vous possedez certains des objets necessaires, je peux vous les prendre, jusqu'a ce que vous ayez completé la liste.";
        menu "Oui",oui,"Non",non;

non:
	  next;
	  mes "["+getarg(0)+"]";
        mes "Tant pis. A une prochaine fois peut-être.";
        close;

oui:
	  next;
        if ((cardinprogress != @card )&&(cardinprogress !=0)) mes "["+getarg(0)+"]";
        if ((cardinprogress != @card )&&(cardinprogress !=0)) mes "Attention, si vous poursuivez, les objets que vous m'avez confié pour la carte precedente seront perdu.";
        if ((cardinprogress != @card )&&(cardinprogress !=0)) menu "Continuer quand meme",-,"Arreter",non;
	  next;
        set item0,@item0;
        set item1,@item1;
        set item2,@item2;
        set item3,@item3;
        set item4,@item4;
        set item5,@item5;
        set item6,@item6;
        set item7,@item7;
        set item8,@item8;
        set amountneeded0,@quantite0; set amount0,0;
        set amountneeded1,@quantite1; set amount1,0;
        set amountneeded2,@quantite2; set amount2,0;
        set amountneeded3,@quantite3; set amount3,0;
        set amountneeded4,@quantite4; set amount4,0;
        set amountneeded5,@quantite5; set amount5,0;
        set amountneeded6,@quantite6; set amount6,0;
        set amountneeded7,@quantite7; set amount7,0;
        set amountneeded8,@quantite8; set amount8,0;
        set cardinprogress,@card;

        L_DonnerObjet2:
                //Verifier que le joueur est a poil pour pas qu'il se fasse niquer ses items equipés.
                deletearray @inventorylist_id[0],128;
                deletearray @inventorylist_amount[0],128;
                deletearray @inventorylist_equip[0],128;
                deletearray @inventorylist_refine[0],128;
                deletearray @inventorylist_card1[0],128;
                deletearray @inventorylist_card2[0],128;
                deletearray @inventorylist_card3[0],128;
                deletearray @inventorylist_card4[0],128;
                getinventorylist;
                set @i,0;
                set @taille,getarraysize(@inventorylist_id);
                Boucle:
                        if (@i >= @taille) goto Suite;
                        if ((@inventorylist_id[@i]==item0)||(@inventorylist_id[@i]==item1)||(@inventorylist_id[@i]==item2)||(@inventorylist_id[@i]==item3)||(@inventorylist_id[@i]==item4)||(@inventorylist_id[@i]==item5)||(@inventorylist_id[@i]==item6)||(@inventorylist_id[@i]==item7)||(@inventorylist_id[@i]==item8))&&((@inventorylist_equip[@i] != 0)||(@inventorylist_refine[@i]!=0)||(@inventorylist_card1[@i]!=0)||(@inventorylist_card2[@i]!=0)||(@inventorylist_card3[@i]!=0)||(@inventorylist_card4[@i]!=0)) goto Warn;
                        set @i,@i+1;
                        goto Boucle;
                Warn:
                        next;
                        mes "["+getarg(0)+"]";
                        mes "Attention, l'item "+getitemname(@inventorylist_id[@i])+" est amélioré ou équipé. Voulez vous tout de meme l'utiliser pour la creation de la card ?";
                        set @i,@i+1;
                        menu "Oui",Boucle,"Non",-;
                        goto Depos;

                Suite:
                  //S'il n'a pas suffisament d'item, ou s'il a juste le compte :
                if (amount0+countitem(item0)<=amountneeded0) goto L_ModifTab0;
            //S'il a trop d'item
                delitem item0,amountneeded0-amount0;
		    set @dejadonne,1;
                set amount0,amountneeded0;
           Back0:
                if (amount1+countitem(item1)<=amountneeded1) goto L_ModifTab1;
                delitem item1,amountneeded1-amount1;
		    set @dejadonne,1;
                set amount1,amountneeded1;
           Back1:
                if (amount2+countitem(item2)<=amountneeded2) goto L_ModifTab2;
                delitem item2,amountneeded2-amount2;
		    set @dejadonne,1;
                set amount2,amountneeded2;
           Back2:
                if (amount3+countitem(item3)<=amountneeded3) goto L_ModifTab3;
                delitem item3,amountneeded3-amount3;
		    set @dejadonne,1;
                set amount3,amountneeded3;
           Back3:
                if (amount4+countitem(item4)<=amountneeded4) goto L_ModifTab4;
                delitem item4,amountneeded4-amount4;
		    set @dejadonne,1;
                set amount4,amountneeded4;
           Back4:
                if (amount5+countitem(item5)<=amountneeded5) goto L_ModifTab5;
                delitem item5,amountneeded5-amount5;
		    set @dejadonne,1;
                set amount5,amountneeded5;
           Back5:
                if (amount6+countitem(item6)<=amountneeded6) goto L_ModifTab6;
                delitem item6,amountneeded6-amount6;
		    set @dejadonne,1;
                set amount6,amountneeded6;
           Back6:
                if (amount7+countitem(item7)<=amountneeded7) goto L_ModifTab7;
                delitem item7,amountneeded7-amount7;
		    set @dejadonne,1;
                set amount7,amountneeded7;
           Back7:
                if (amount8+countitem(item8)<=amountneeded8) goto L_ModifTab8;
                delitem item8,amountneeded8-amount8;
		    set @dejadonne,1;
                set amount8,amountneeded8;
           goto L_FinRecup;

L_ModifTab0:
        set amount0,amount0+countitem(item0);
        delitem item0,countitem(item0);
	  set @dejadonne,1;
        goto Back0;
L_ModifTab1:
        set amount1,amount1+countitem(item1);
        delitem item1,countitem(item1);
	  set @dejadonne,1;
        goto Back1;
L_ModifTab2:
        set amount2,amount2+countitem(item2);
        delitem item2,countitem(item2);
	  set @dejadonne,1;
        goto Back2;
L_ModifTab3:
        set amount3,amount3+countitem(item3);
        delitem item3,countitem(item3);
	  set @dejadonne,1;
        goto Back3;
L_ModifTab4:
        set amount4,amount4+countitem(item4);
        delitem item4,countitem(item4);
	  set @dejadonne,1;
        goto Back4;
L_ModifTab5:
        set amount5,amount5+countitem(item5);
        delitem item5,countitem(item5);
	  set @dejadonne,1;
        goto Back5;
L_ModifTab6:
        set amount6,amount6+countitem(item6);
        delitem item6,countitem(item6);
	  set @dejadonne,1;
        goto Back6;
L_ModifTab7:
        set amount7,amount7+countitem(item7);
        delitem item7,countitem(item7);
	  set @dejadonne,1;
        goto Back7;
L_ModifTab8:
        set amount8,amount8+countitem(item8);
        delitem item8,countitem(item8);
	  set @dejadonne,1;
        goto L_FinRecup;

L_FinRecup:
		next;
        if (amount0!=amountneeded0) goto L_ListeitemsManquants;
        if (amount1!=amountneeded1) goto L_ListeitemsManquants;
        if (amount2!=amountneeded2) goto L_ListeitemsManquants;
        if (amount3!=amountneeded3) goto L_ListeitemsManquants;
        if (amount4!=amountneeded4) goto L_ListeitemsManquants;
        if (amount5!=amountneeded5) goto L_ListeitemsManquants;
        if (amount6!=amountneeded6) goto L_ListeitemsManquants;
        if (amount7!=amountneeded7) goto L_ListeitemsManquants;
        if (amount8!=amountneeded8) goto L_ListeitemsManquants;
        goto L_ToutLesitems;

L_ToutLesitems:
          getitem @card,1; //Invocation de la card.
        set cardinprogress,0;
        set amount0,0;
        set amount1,0;
        set amount2,0;
        set amount3,0;
        set amount4,0;
        set amount5,0;
        set amount6,0;
        set amount7,0;
        set amount8,0;

        mes "["+getarg(0)+"]";
        mes "Félicitations !";
        close;

Depos:
        mes "["+getarg(0)+"]";
        mes "Veuillez deposer votre "+getitemname(@inventorylist_id[@i-1])+" dans le Storage.";
	  next;
        close2;
        openstorage;
        end;
}
